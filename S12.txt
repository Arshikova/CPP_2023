#include <iostream>

using namespace std;
int main(){
setlocale(0, "Russian");
int s = 0;
for (int a = 1; a <=1000; a+=4){ //перебрали числа от 1 до 1000 с шагом 4 
    if ( a%7==0){ // нашли числа,которые делятся на 7 без остатка 
        s+=a; // сумма всех найденных чисел
    }
}
cout << "Сумма всех чисел, которые делятся на 7 без остатка:"<< s<< endl;
return 0;
}
